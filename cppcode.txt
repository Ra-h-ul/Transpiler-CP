	#include <iostream>
	#include <iomanip>
	#include <string>


	template <typename T>
	void _format_output(std::ostream& out, const T& str) {
		// Pad the string with spaces so that it is at least 20 characters wide.
		out << str;
	}
		

	auto main() -> int {
	auto x = 10;
	auto y = 6;
	auto z = x+y;
	std::string s3 = std::to_string(z);
	_format_output(std::cout, s3); 
	std::cout << std::endl;
	if (z > 0) {
	_format_output(std::cout, z + " is greater that zero");
	std::cout << std::endl;
		}

		
	return 0;
	}
